	int max = niz[0];
	for (int i = 0; i < 10; i++) {
		if (niz[i] > max) {
			max = niz[i];
		}
	}
	return max;
}

	int min = niz[0];

		if (niz[i] < min) {

		}
	}
	return min;
}

#include <iostream>

struct Niz {
	int duzina;
	int* elementi;
};



void Random() {
	srand(time(0));
}

void Inicijalizaja(Niz niz) {
	for (int i = 0; i < niz.duzina; i++)
	{
		niz.elementi[i] = rand();
	}
}

void Ispis(Niz niz) {
	for (int i = 0; i < niz.duzina; i++) {
		cout << niz.elementi[i] << " ";
	}
}


int max(Niz niz) {
	int max = niz.elementi[0];
	for (int i = 0; i < niz.duzina; i++) {
		if (niz.elementi[i] > max) {
			max = niz.elementi[i];
		}
	}
	return max;
}


int min(Niz niz) {
	int min = niz.elementi[0];
	for (int i = 0; i < niz.duzina; i++) {
		if (niz.elementi[i] < min) {
			min = niz.elementi[i];
		}
	}
	return min;
}

void descending(Niz niz) {
	for (int i = 0; i < niz.duzina; i++) {
		for (int j = 0; j < niz.duzina; j++) {

				pholder = niz.elementi[i];
				niz.elementi[i] = niz.elementi[j];
				niz.elementi[j] = pholder;
			}

		}

	}
	 for (int i = 0; i < niz.duzina; i++) {
		cout << niz.elementi[i] << " ";
	}
}
